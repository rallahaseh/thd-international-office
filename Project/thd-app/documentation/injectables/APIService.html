import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { map } from 'rxjs/operators';
import { ENDPOINTS, API_BASE_URL } from './api-service-endpoints';

@Injectable({ providedIn: 'root' })
export class APIService {

    constructor(private http: HttpClient) { }

    get<T>(endpoint: string, params: Object = {}) {
        return this.http.get<T>(this.getURL(endpoint, params)).pipe(
            map((response: T) => response)
        );
    }

    post<T>(endpoint: string, params: Object = {}, data: Object = {}) {
        return this.http.post<T>(this.getURL(endpoint, params), data).pipe(
            map((response: T) => response)
        );
    }

    put<T>(endpoint: string, params: Object = {}, data: Object = {}) {
        return this.http.put<T>(this.getURL(endpoint, params), data).pipe(
            map((response: T) => response)
        );
    }

    delete<T>(endpoint: string, params: Object = {}, data: Object = {}) {
        return this.http.delete<T>(this.getURL(endpoint, params), data).pipe(
            map((response: T) => response)
        );
    }

    getURL(endpoint: string, params: Object) {
        return this.getParams(endpoint, params);
    }

    getParams(url: string, params: Object) {
        Object.entries(params).forEach(
            ([key, value]) => {
                url = url.replace(new RegExp("\\{" + key + "\\}", "gm"), value);
            }
        );
        return API_BASE_URL + url;
    }
}
